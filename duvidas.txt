CREATE TABLE Appointment (
	id PRIMARY KEY INTEGER NOT NULL,
	date DATE NOT NULL,
	user_id INTEGER NOT NULL,
	provider_id INTEGER NOT NULL,
  created_at DATE NOT NULL,
  updated_at DATE NOT NULL,
	CONSTRAINT PK_APPOINTMENT
		PRIMARY KEY (id),
  CONSTRAINT appointment_user_id
    FOREIGN KEY (user_id)
    REFERENCES users(id)
    ON UPDATE CASCADE
    ON DELETE CASCADE,
  CONSTRAINT appointment_provider_id
    FOREIGN KEY (provider_id)
    REFERENCES users(id)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);
